void only1(vector<set<int>> &vs, set<int> &s1){
    vector<set<int>>::iterator v;
    set<int>::iterator p,t;
    multiset<int> multiaux;
    set<int> setaux;
    for(v=vs.begin();v!=vs.end();++v){
            for(p=(*v).begin();p!=(*v).end();++p){
                multiaux.insert(*p);
                if(!setaux.count(*p)){
                    setaux.insert(*p);
                }
        }
    }
    for(t=setaux.begin();t!=setaux.end();++t){
        if(multiaux.count(*t)==1){
                s1.insert(*t);
            }
        }
}
